-------------------------------------------------

enable secret <CCNA>>

 no shutdown

interface g0/1
 ip address 10.0.12.1 255.255.255.252
 no shutdown

interface g0/2
 ip address 203.0.113.1 255.255.255.252
 no shutdown

interface loopback0
 ip address 1.1.1.1 255.255.255.255
